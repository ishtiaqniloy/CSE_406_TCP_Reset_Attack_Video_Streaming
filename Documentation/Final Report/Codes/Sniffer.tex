#!/usr/bin/python3
#running command: python Sniffer.py victimIP

import sys  #for command line argument
from scapy.all import *
from scapy.layers.inet import *
import scapy

victim_ip = sys.argv[1]

def func(pkt):
    print("PKT INFO:")
    # print(pkt.__class__)
    print(pkt.summary())

    print(pkt[IP].src)
    print(pkt[IP].dst)

    print(pkt[TCP].sport)
    print(pkt[TCP].dport)
    print(pkt[TCP].ack)

    print()

pkt = sniff(filter="tcp and src host " + victim_ip, 
    prn=func, store=0)
